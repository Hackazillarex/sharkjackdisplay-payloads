#!/bin/bash
# Title: Simple Client Info
# Author: Hak5
# Description: Hosts a DHCP server and displays the hostname, IP address and MAC address of each connected client.
# Usage: Run payload. Plug Shark Jack into target. Once the target obtains a DHCP lease, the info will display.
LED SETUP
SCREEN_WRITE "Starting DHCP"
NETMODE DHCP_SERVER
WAIT_FOR_LINK
WAIT_FOR_CLIENT
LED FINISH
SCREEN_WRITE $CLIENT_HOSTNAME
SCREEN_WRITE $CLIENT_IP
SCREEN_WRITE $CLIENT_MAC
